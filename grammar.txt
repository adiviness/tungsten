
start : block $

block : statements

statments : statement statements
    |

statement : declaration NEWLINE
    | NEWLINE

declaration : IDENTIFIER IDENTIFIER ASSIGN expression
    | IDENTIFIER ASSIGN expression

expression : val
    | val binary_op expression
    | unary_op val
    | unary_op val binary_op expression


binary_op : PLUS
    | MINUS
    | MULTIPLY
    | DIVIDE
    | OR
    | AND
   

unary_op : MINUS
    | NOT

val : INTEGER
    | FLOAT
    | TRUE
    | FALSE
    | IDENTIFIER
