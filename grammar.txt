
start : block $

block : statements

statments : statement statements
    |

statement : declaration

declaration : IDENTIFIER IDENTIFIER ASSIGN expression

expression : val
    | expression binary_op val
    | unary_op val

binary_op : PLUS
    | MINUS
    | MULTIPLY
    | DIVIDENTIFIERE

unary_op : MINUS
    | NOT

val : INTEGER
    | FLOAT
    | IDENTIFIER
